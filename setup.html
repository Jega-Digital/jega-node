1. Integration & Setup (Node.js)

Wrap your standard fetch requests with the x402 SDK. This will automatically negotiate the paywall challenge, sign the micro-transaction, and return the data.

import { wrapFetchWithPayment, x402Client } from "@x402/fetch";
import { privateKeyToAccount } from "viem/accounts";
import { base } from "viem/chains";

// 1. Initialize your agent's secure wallet
const account = privateKeyToAccount("0x_YOUR_AGENT_PRIVATE_KEY");
const client = new x402Client();

// 2. Register the payment scheme for Base Mainnet
client.registerExactEvmScheme({ signer: account, chainId: base.id });

// 3. Wrap the standard fetch API
const fetchWithPayment = wrapFetchWithPayment(fetch, client);

// 4. Execute. The SDK handles the 402 challenge and streams USDC.
const response = await fetchWithPayment("https://api.jega.digital/api/compress", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({ 
        raw_text: "This   is a \n\n very bloated \t string of text." 
    })
});

const data = await response.json();
console.log(data);
